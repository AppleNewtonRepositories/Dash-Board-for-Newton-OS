SetPartFrameSlot('DoNotInstall, func() //return TRUE if not strongarm	begin		local g := Gestalt(kGestalt_SystemInfo);		if g.cputype = 'strongarm then			begin				return NIL;			end		else			begin				GetRoot():Notify(kNotifyALert, TotalClone("Dash Board Prefs"), TotalClone("Sorry, Dash Board Prefs can only be installed on MessagePad 2000 or MessagePad 2100 machines."));				return TRUE;			end;	end;);
